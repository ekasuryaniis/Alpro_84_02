Deklarasi Variabel :               
    pivot, min, cari, titip, N bertipe int           
    data bertipe array of int pada variabel data           
{               
    FOR(pivot=0; pivot<N-1; pivot++)           
    {           
        min = pivot       
        FOR(cari=pivot+1; cari<N; cari++)       
        {       
            IF(data[cari] < data[min])   
                min = cari
        }       
        titip = data[pivot]       
        data[pivot] = data[min]       
        data[min] = titip       
    }           
}
